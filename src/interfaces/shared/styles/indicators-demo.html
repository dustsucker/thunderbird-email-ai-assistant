5. Usage Examples

TypeScript/JavaScript

// Create an indicator component
const indicator = new LowConfidenceIndicator();

// Create an icon indicator
const iconElement = indicator.createIcon(0.65, 'list');
document.body.appendChild(iconElement);

// Create a badge indicator
const badgeElement = indicator.createBadge(0.75, 'view');
document.body.appendChild(badgeElement);

// Create with options
const customIndicator = indicator.create({
  confidence: 0.55,
  type: 'badge',
  placement: 'list',
  tooltip: 'Low confidence - requires review',
  animate: true,
  isFlagged: true
});

// Mark an email as flagged
indicator.markEmailAsFlagged(emailElement, true);

// Create notification banner
const banner = indicator.createNotificationBanner({
  count: 3,
  message: '3 emails require review',
  actionText: 'Review Now',
  onAction: () => console.log('Review clicked'),
  autoDismiss: 5000
});
document.body.appendChild(banner);

Visibility Control

// Set visibility configuration
indicator.setVisibilityConfig({
  showIndicators: true,
  showIcons: true,
  showBadges: true,
  showNotifications: true
});

// Or control via body classes
document.body.classList.add('indicators-hidden');         // Hide all
document.body.classList.add('indicators-icons-only');     // Icons only
document.body.classList.add('indicators-badges-only');    // Badges only
document.body.classList.add('indicators-no-notifications'); // No notifications

6. Accessibility Features

  • ARIA Labels: All indicators have proper aria-label describing confidence level and status
  • Keyboard Navigation: Focus-visible styles for keyboard users
  • High Contrast: Enhanced borders in high-contrast mode
  • Reduced Motion: Animations respect prefers-reduced-motion
  • Screen Reader Support: role="img" and descriptive labels for screen readers
